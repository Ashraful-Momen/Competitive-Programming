## üèóÔ∏è **Object-Oriented Programming (OOP)**

### **üì¶ Classes & Objects:**
```cpp
class Student {
private:
    string name;
    int age;
    vector<int> grades;

public:
    // Constructor
    Student(string n, int a) : name(n), age(a) {}
    
    // Default constructor
    Student() : name("Unknown"), age(0) {}
    
    // Destructor
    ~Student() {
        cout << "Student " << name << " destroyed" << endl;
    }
    
    // Getter methods
    string getName() const { return name; }
    int getAge() const { return age; }
    
    // Setter methods
    void setName(string n) { name = n; }
    void setAge(int a) { age = a; }
    
    // Other methods
    void addGrade(int grade) {
        grades.push_back(grade);
    }
    
    double getAverage() const {
        if (grades.empty()) return 0.0;
        int sum = 0;
        for (int grade : grades) {
            sum += grade;
        }
        return static_cast<double>(sum) / grades.size();
    }
    
    void displayInfo() const {
        cout << "Name: " << name << ", Age: " << age;
        cout << ", Average: " << getAverage() << endl;
    }
};

// Usage
int main() {
    Student s1("Alice", 20);
    Student s2;                          // Uses default constructor
    
    s1.addGrade(85);
    s1.addGrade(92);
    s1.addGrade(78);
    
    s1.displayInfo();
    
    return 0;
}
```

### **üîó Inheritance:**

```cpp
// Base class
class Animal {
protected:
    string name;
    int age;

public:
    Animal(string n, int a) : name(n), age(a) {}
    
    virtual void makeSound() {          // Virtual function
        cout << name << " makes a sound" << endl;
    }
    
    void sleep() {
        cout << name << " is sleeping" << endl;
    }
};

// Derived class
class Dog : public Animal {
private:
    string breed;

public:
    Dog(string n, int a, string b) : Animal(n, a), breed(b) {}
    
    void makeSound() override {         // Override virtual function
        cout << name << " barks: Woof!" << endl;
    }
    
    void wagTail() {
        cout << name << " is wagging tail" << endl;
    }
};

// Usage
int main() {
    Dog myDog("Buddy", 3, "Golden Retriever");
    myDog.makeSound();                  // "Buddy barks: Woof!"
    myDog.sleep();                      // Inherited method
    myDog.wagTail();                    // Dog-specific method
    
    return 0;
}
```

```cpp
}
```

```cpp
void divideNumbers() {

    }

    try {
        throw CustomException("This is a custom error!");
    }
    catch (const CustomException& e) {

}
```

```cpp
// Generic function template
template<typename T>
T add(T a, T b) {
    return a + b;
}

// Template with multiple parameters
template<typename T, typename U>
auto multiply(T a, U b) -> decltype(a * b) {
    return a * b;
}

// Template specialization
template<>
string add<string>(string a, string b) {
    return a + " " + b;
}

// Usage
int main() {
    cout << add(5, 3) << endl;           // int version: 8
    cout << add(5.5, 3.2) << endl;      // double version: 8.7
    cout << add<string>("Hello", "World") << endl; // specialized: "Hello World"
    
    auto result = multiply(5, 3.14);    // int * double = double
    
    return 0;
}
```

```cpp
class Container {
private:
    vector<T> items;

    void add(const T& item) {
```

## üêõ **Common Mistakes & Best Practices**

### **‚ùå Common Mistakes:**
```cpp
// 1. Array bounds violation
int arr[5];
arr[10] = 100;                       // ‚ùå Out of bounds!

// 2. Memory leaks
int* ptr = new int(42);
// delete ptr;                       // ‚ùå Forgot to delete!

// 3. Uninitialized variables
int x;                               // ‚ùå Uninitialized!
cout << x << endl;                   // Undefined behavior

// 4. Infinite loops
// while (true) { }                  // ‚ùå No exit condition!

// 5. Integer division
int result = 5 / 2;                  // ‚ùå Result is 2, not 2.5!
```

### **‚úÖ Best Practices:**
```cpp
// 1. Always initialize variables
int x = 0;
double y = 0.0;
bool flag = false;

// 2. Use const when possible
const int MAX_SIZE = 100;
void printValue(const string& str) { /* ... */ }

// 3. Use meaningful names
int studentAge;                      // ‚úÖ Good
int a;                               // ‚ùå Bad

// 4. Use RAII (Resource Acquisition Is Initialization)
vector<int> v;                       // ‚úÖ Automatic cleanup
unique_ptr<int> ptr = make_unique<int>(42); // ‚úÖ Automatic cleanup

// 5. Prefer algorithms over raw loops
// Instead of manual loop:
// for (int i = 0; i < v.size(); i++) {
//     if (v[i] == target) return i;
// }

// Use STL algorithm:
auto it = find(v.begin(), v.end(), target);

// 6. Use range-based for loops
for (const auto& item : container) { /* ... */ }
```
