### **üî¢ Essential Functions:**
```cpp
// GCD & LCM
ll gcd(ll a, ll b) { return b ? gcd(b, a % b) : a; }
ll lcm(ll a, ll b) { return a / gcd(a, b) * b; }

// Power with mod
ll power(ll a, ll b, ll mod = MOD) {
    ll res = 1;
    a %= mod;
    while(b > 0) {
        if(b & 1) res = (res * a) % mod;
        a = (a * a) % mod;
        b >>= 1;
    }
    return res;
}

// Modular inverse
ll modinv(ll a, ll mod = MOD) {
    return power(a, mod - 2, mod);
}

// Check prime (optimized)
bool isPrime(ll n) {
    if(n < 2) return false;
    if(n == 2 || n == 3) return true;
    if(n % 2 == 0 || n % 3 == 0) return false;
    for(ll i = 5; i * i <= n; i += 6) {
        if(n % i == 0 || n % (i + 2) == 0) return false;
    }
    return true;
}

// Count divisors
int countDivisors(ll n) {
    int cnt = 0;
    for(ll i = 1; i * i <= n; i++) {
        if(n % i == 0) {
            cnt += (i * i == n) ? 1 : 2;
        }
    }
    return cnt;
}
```

```cpp
int binarySearch(vi& arr, int target) {

    while(left <= right) {
```

### **üö∂ DFS & BFS:**
```cpp
// DFS
vb visited(n + 1, false);
void dfs(int node) {
    visited[node] = true;
    // Process node
    for(int neighbor : adj[node]) {
        if(!visited[neighbor]) {
            dfs(neighbor);
        }
    }
}

// BFS
void bfs(int start) {
    queue<int> q;
    vb visited(n + 1, false);
    
    q.push(start);
    visited[start] = true;
    
    while(!q.empty()) {
        int node = q.front();
        q.pop();
        
        // Process node
        for(int neighbor : adj[node]) {
            if(!visited[neighbor]) {
                visited[neighbor] = true;
                q.push(neighbor);
            }
        }
    }
}

// Connected components
int countComponents() {
    vb visited(n + 1, false);
    int components = 0;
    
    rep1(i, n) {
        if(!visited[i]) {
            dfs(i);
            components++;
        }
    }
    return components;
}
```

### **üõ£Ô∏è Shortest Path:**
```cpp
// Dijkstra's Algorithm
vi dijkstra(int start) {
    vi dist(n + 1, INF);
    priority_queue<pii, vector<pii>, greater<pii>> pq;
    
    dist[start] = 0;
    pq.push({0, start});
    
    while(!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        
        if(d > dist[u]) continue;
        
        for(auto [v, w] : wadj[u]) {
            if(dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}
```

### **üíé Classical DP:**
```cpp
// Fibonacci
vi dp(n + 1, -1);
int fib(int n) {
    if(n <= 1) return n;
    if(dp[n] != -1) return dp[n];
    return dp[n] = fib(n-1) + fib(n-2);
}

// Coin Change
int coinChange(vi& coins, int amount) {
    vi dp(amount + 1, INF);
    dp[0] = 0;
    
    rep1(i, amount) {
        for(int coin : coins) {
            if(i >= coin && dp[i - coin] != INF) {
                dp[i] = min(dp[i], dp[i - coin] + 1);
            }
        }
    }
    return dp[amount] == INF ? -1 : dp[amount];
}

// Longest Increasing Subsequence
int lis(vi& arr) {
    vi dp;
    for(int x : arr) {
        auto it = lower_bound(all(dp), x);
        if(it == dp.end()) dp.pb(x);
        else *it = x;
    }
    return sz(dp);
}

// 0/1 Knapsack
int knapsack(vi& wt, vi& val, int W) {
    int n = sz(wt);
    vector<vi> dp(n + 1, vi(W + 1, 0));
    
    rep1(i, n) {
        rep1(w, W) {
            if(wt[i-1] <= w) {
                dp[i][w] = max(dp[i-1][w], dp[i-1][w-wt[i-1]] + val[i-1]);
            } else {
                dp[i][w] = dp[i-1][w];
            }
        }
    }
    return dp[n][W];
}
```

```cpp
// KMP Pattern Matching
vi computeLPS(string pattern) {
    int m = sz(pattern);
    vi lps(m, 0);
    int len = 0, i = 1;
    
    while(i < m) {
        if(pattern[i] == pattern[len]) {
            len++;
            lps[i] = len;
            i++;
        } else {
            if(len != 0) len = lps[len - 1];
            else { lps[i] = 0; i++; }
        }
    }
    return lps;
}

vi KMP(string text, string pattern) {
    vi lps = computeLPS(pattern);
    vi result;
    int i = 0, j = 0;
    
    while(i < sz(text)) {
        if(pattern[j] == text[i]) { j++; i++; }
        
        if(j == sz(pattern)) {
            result.pb(i - j);
            j = lps[j - 1];
        } else if(i < sz(text) && pattern[j] != text[i]) {
            if(j != 0) j = lps[j - 1];
            else i++;
        }
    }
    return result;
}
```

```cpp
// Two Sum (sorted array)
pii twoSum(vi& arr, int target) {
    int left = 0, right = sz(arr) - 1;
    while(left < right) {
        int sum = arr[left] + arr[right];
        if(sum == target) return {left, right};
        else if(sum < target) left++;
        else right--;
    }
    return {-1, -1};
}

// Remove duplicates
int removeDuplicates(vi& arr) {
    if(sz(arr) == 0) return 0;
    int j = 0;
    rep1(i, sz(arr) - 1) {
        if(arr[i] != arr[j]) {
            j++;
            arr[j] = arr[i];
        }
    }
    return j + 1;
}
```

### **ü™ü Sliding Window:**
```cpp
// Maximum sum subarray of size k
int maxSumSubarray(vi& arr, int k) {
    if(sz(arr) < k) return -1;
    
    int windowSum = 0;
    rep(i, k) windowSum += arr[i];
    
    int maxSum = windowSum;
    for(int i = k; i < sz(arr); i++) {
        windowSum = windowSum - arr[i - k] + arr[i];
        maxSum = max(maxSum, windowSum);
    }
    return maxSum;
}

// Longest substring without repeating characters
int lengthOfLongestSubstring(string s) {
    map<char, int> charIndex;
    int maxLen = 0, start = 0;
    
    rep(end, sz(s)) {
        if(charIndex.count(s[end])) {
            start = max(start, charIndex[s[end]] + 1);
        }
        charIndex[s[end]] = end;
        maxLen = max(maxLen, end - start + 1);
    }
    return maxLen;
}
```
