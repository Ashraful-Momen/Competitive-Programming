#include <bits/stdc++.h>
using namespace std;

int main(){

    int row,col;
    cin >> row >>col;

    vector<vector <int>> arr(row, vector <int>(col));

     // Matrix Input
    for (int i = 0; i < row; i++) {
        for (int j = 0; j < col; j++) {
            cin >> arr[i][j];
        }
    }

    cout << "Matrix is:\n";

    // Matrix Output
    for (int i = 0; i < row; i++) {
        for (int j = 0; j < col; j++) {
            cout << arr[i][j] << " ";
        }
        cout << "\n";
    }





    return 0;
}
