using namespace std;

int main() {
    int arr[2][3][4]; // 2 blocks, each 3x4

    // Initializing the array
    int value = 1;
    for (int i = 0; i < 2; i++) { // blocks
        for (int j = 0; j < 3; j++) { // rows
            for (int k = 0; k < 4; k++) { // columns
                arr[i][j][k] = value++;
            }
        }
    }

    // Printing all elements
    for (int i = 0; i < 2; i++) {
        cout << "Block " << i << ":\n";
        for (int j = 0; j < 3; j++) {
            for (int k = 0; k < 4; k++) {
                cout << arr[i][j][k] << "\t";
            }
            cout << endl;
        }
        cout << endl;
    }

    return 0;
}
